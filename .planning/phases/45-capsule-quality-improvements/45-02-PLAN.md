# Plan 45-02: Calls Section + Resolved-Imports Section

## Objective

Add two new capsule sections: `calls` (P1) listing outgoing call references with resolved file:line, and `resolved-imports` (P1) listing workspace-internal import resolutions. The `calls` section is built in `buildCodeCapsule` from `callSitesRaw` (callee-name only pre-enrichment). After `enrichWorkspaceRecords` Pass 2 resolves callSites, a post-enrichment step rebuilds the calls section with file:line. The `resolved-imports` section is built in `buildWorkspaceFileRecord` where `workspaceFilePaths` is available. Create test suites for both.

## must_haves

- `callsSection` (P1) in `buildCodeCapsule` (both compression-core.cjs and tree-sitter-worker.cjs), populated from `callSitesRaw` showing `calleeName` pre-enrichment
- After `enrichWorkspaceRecords` Pass 2, rebuild calls section with resolved `resolvedFile:resolvedLine` format
- `resolvedImportsSection` (P1) added to `capsuleBase.sections` in `buildWorkspaceFileRecord` after `dependencies[]` is built
- Both sections only appear when non-empty (existing flatMap filter)
- `test/capsule-calls.test.cjs` and `test/capsule-imports.test.cjs` exist and pass

After enrichment Pass 2 resolves `fileRecord.callSites[]` with `resolvedFile` and `resolvedLine`, rebuild the `calls` section in `capsuleBase.sections`:

Find the `enrichWorkspaceRecords` function in `mesh-core/src/workspace-operations.js`. After Pass 2 sets `fileRecord.callSites = resolvedCallSites` (or equivalent), add:

```javascript
// Rebuild calls section with resolved file:line after enrichment
const MAX_CALLS_SECTION = 30;
const callsSectionItems = [];
for (const cs of fileRecord.callSites) {
  if (callsSectionItems.length >= MAX_CALLS_SECTION) break;
  const location = cs.resolvedFile
    ? `${cs.resolvedFile}:${cs.resolvedLine}`
    : `line ${cs.callerLine}`;
  callsSectionItems.push({ text: `${cs.calleeName} → ${location}`, priority: "P1" });
}
if (callsSectionItems.length > 0) {
  const sections = fileRecord.capsuleBase?.sections;
  if (Array.isArray(sections)) {
    const existingIdx = sections.findIndex(s => s.name === 'calls');
    const updatedSection = { name: 'calls', priority: 'P1', items: callsSectionItems };
    if (existingIdx >= 0) {
      sections[existingIdx] = updatedSection;
    } else {
      sections.push(updatedSection);
    }
  }
  if (fileRecord.capsuleCache?.capsule?.sections) {
    const cacheSections = fileRecord.capsuleCache.capsule.sections;
    const cacheIdx = cacheSections.findIndex(s => s.name === 'calls');
    const updatedSection = { name: 'calls', priority: 'P1', items: callsSectionItems };
    if (cacheIdx >= 0) {
      cacheSections[cacheIdx] = updatedSection;
    } else {
      cacheSections.push(updatedSection);
    }
  }
}
```

Add this block inside the Pass 2 `mapWithConcurrency` or `for...of` loop, after the callSites resolution step, for each file record.
</action>

const { describe, it } = require('node:test');
const assert = require('node:assert/strict');
const { buildWorkspaceFileRecord } = require('../mesh-core/src/compression-core.cjs');

describe('capsule calls section — CAP-02', () => {
  it('given a JS file calling another function, when built, then capsule has calls section', async () => {
    const record = await buildWorkspaceFileRecord('app.js', 'function main() {\n  login();\n  logout();\n}\n');
    const sections = record.capsuleBase?.sections || [];
    const callsSection = sections.find(s => s.name === 'calls');
    assert.ok(callsSection, 'calls section should exist');
    assert.ok(callsSection.items.length >= 1, 'calls section should have entries');
    const texts = callsSection.items.map(i => i.text);
    assert.ok(texts.some(t => t.includes('login') || t.includes('logout')), 'calls should reference login or logout');
  });

  it('given a JS file with no outgoing calls, when built, then no calls section', async () => {
    const record = await buildWorkspaceFileRecord('constants.js', 'const MAX = 100;\nconst PI = 3.14;\n');
    const sections = record.capsuleBase?.sections || [];
    const callsSection = sections.find(s => s.name === 'calls');
    assert.ok(!callsSection || callsSection.items.length === 0, 'calls section should be absent or empty for pure constants');
  });
});
