# Plan 29-03: Agent Token Backend

## Objective

Build the server-side infrastructure for the local terminal agent: (1) agent token creation/verification functions in `secure-db.js`, (2) REST endpoints in `auth.routes.js` to generate and check token/status, (3) a new `/terminal-agent` WebSocket endpoint in `terminal.routes.js` that accepts agent connections and proxies them to waiting browser sessions.

## must_haves

- `createAgentToken(userId)` and `findAgentToken(rawToken)` exported from `secure-db.js`
- `POST /api/v1/terminal/agent-token` creates or returns existing token for authenticated user
- `GET /api/v1/terminal/agent-status` returns whether agent is currently connected for authenticated user
- `/terminal-agent` WebSocket endpoint authenticates agents by token, stores connection in-memory per userId
- `/terminal` WebSocket endpoint checks for active agent connection and proxies if found; falls back to EC2 shell if no agent

```javascript
/**
 * Creates a long-lived agent token for a user.
 * Stored in the sessions table with type='agent-token'.
 * Returns existing token if one already exists (idempotent).
 *
 * @param {string} userId
 * @returns {Promise<string>} raw token
 */
async function createAgentToken(userId) {
  const uid = String(userId || '').trim();
  if (!uid) throw new Error('userId is required');

  // Check for existing token first (idempotent)
  const existing = await findAgentTokenByUserId(uid);
  if (existing) return existing;

  const token = crypto.randomBytes(32).toString('hex');
  const hashId = hashSessionToken(token);
  const doc = {
    id: hashId,
    type: 'agent-token',
    userId: uid,
    createdAt: toIsoNow(),
  };

  if (!dynamoEnabled) {
    memSessions.set(hashId, doc);
    return token;
  }

  await getDocClient().send(new PutCommand({ TableName: DYNAMO_SESSIONS_TABLE, Item: doc }));
  return token;
}

/**
 * Finds a user's existing agent token hash and returns it as a raw token.
 * Used internally by createAgentToken for idempotency.
 * NOTE: We cannot reverse-hash, so this scans for existing agent-token by userId.
 *
 * @param {string} userId
 * @returns {Promise<string|null>}
 */
async function findAgentTokenByUserId(userId) {
  const uid = String(userId || '').trim();
  if (!uid) return null;

  if (!dynamoEnabled) {
    for (const [, doc] of memSessions) {
      if (doc.type === 'agent-token' && doc.userId === uid) {
        // We stored the raw token in a separate field for in-memory mode
        return doc.rawToken || null;
      }
    }
    return null;
  }

  // DynamoDB: query by userId via GSI (userId-index already exists on sessions table per existing code)
  try {
    const result = await getDocClient().send(new QueryCommand({
      TableName: DYNAMO_SESSIONS_TABLE,
      IndexName: 'userId-index',
      KeyConditionExpression: 'userId = :uid',
      FilterExpression: '#t = :type',
      ExpressionAttributeNames: { '#t': 'type' },
      ExpressionAttributeValues: { ':uid': uid, ':type': 'agent-token' },
      Limit: 1,
    }));
    // Cannot return original token from hash — generate fresh on each lookup from DB
    // For idempotency in DynDB mode, we rely on the token being stored with rawToken field
    const item = result.Items?.[0];
    return item?.rawToken || null;
  } catch {
    return null;
  }
}

/**
 * Verifies an agent token and returns the associated userId.
 *
 * @param {string} rawToken
 * @returns {Promise<string|null>} userId or null if invalid
 */
async function findAgentToken(rawToken) {
  const token = String(rawToken || '').trim();
  if (!token) return null;
  const hashId = hashSessionToken(token);

  if (!dynamoEnabled) {
    const doc = memSessions.get(hashId);
    return (doc?.type === 'agent-token') ? doc.userId : null;
  }

  try {
    const result = await getDocClient().send(new GetCommand({
      TableName: DYNAMO_SESSIONS_TABLE,
      Key: { id: hashId },
    }));
    const doc = result.Item;
    return (doc?.type === 'agent-token') ? doc.userId : null;
  } catch {
    return null;
  }
}
```

**Fix for idempotency with rawToken:** In `createAgentToken`, store `rawToken: token` in the doc so `findAgentTokenByUserId` can return it:

    type: 'agent-token',
    userId: uid,
    rawToken: token,   // stored for idempotency lookup; hashed id is the secure key
    createdAt: toIsoNow(),
  };
```

/**
 * POST /api/v1/terminal/agent-token
 * Creates (or returns existing) long-lived agent token for the authenticated user.
 * Used by the connect dialog to generate the npx mesh-local command.
 */
router.post('/terminal/agent-token', async (req, res) => {
  try {
    const user = await resolveAuthUserFromRequest(req);
    if (!user) return res.status(401).json({ error: { code: 'UNAUTHORIZED', message: 'Not authenticated' }, data: null });
    const token = await createAgentToken(user.id || user.userId || user.email);
    const serverUrl = `${req.protocol}://${req.get('host')}`;
    return res.json({
      data: {
        token,
        command: `npx mesh-local --token=${token} --server=${serverUrl}`,
        meshUrl: `mesh://launch-agent?token=${token}&server=${encodeURIComponent(serverUrl)}`,
      },
      error: null,
    });
  } catch (err) {
    return res.status(500).json({ error: { code: 'AGENT_TOKEN_ERROR', message: 'Failed to generate agent token' }, data: null });
  }
});

/**
 * GET /api/v1/terminal/agent-status
 * Returns whether a local agent is currently connected for the authenticated user.
 * Polled by the browser connect dialog (every 1.5s) to detect when agent connects.
 */
router.get('/terminal/agent-status', async (req, res) => {
  try {
    const user = await resolveAuthUserFromRequest(req);
    if (!user) return res.status(401).json({ error: { code: 'UNAUTHORIZED', message: 'Not authenticated' }, data: null });
    const userId = user.id || user.userId || user.email;
    // agentConnections is a shared Map populated by terminal.routes.js
    const isConnected = req.app.locals.agentConnections?.has(userId) ?? false;
    return res.json({ data: { connected: isConnected }, error: null });
  } catch (err) {
    return res.status(500).json({ error: { code: 'STATUS_ERROR', message: 'Failed to check agent status' }, data: null });
  }
});
```
